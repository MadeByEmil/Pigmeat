62  public static object ConvertFromJson(JObject element)
63  {
64  switch (element.Type)
65  {
66  case JTokenType.Array:
67  var array = new ScriptArray();
68  foreach (var nestedElement in element.Descendants())
69  {
70  array.Add(element.ToObject<object>());
71  }
72  return array;
73  case JTokenType.String:
74  return element.ToString();
75  case JTokenType.Integer:
76  return element.ToObject<int>();
77  case JTokenType.Float:
78  return element.ToObject<double>();
79  case JTokenType.Boolean:
80  return element.ToObject<bool>();
81  case JTokenType.Object:
82  var obj = new ScriptObject();
83  foreach (var prop in element.Properties())
84  {
85  var Value = prop.Value.ToObject<object>();
86 
87  try
88  {
89  Value.ToString(); // If an object is null, a NullReferenceException occurs
90  }
91  catch(NullReferenceException)
92  {
93  Value = "null"; // Turn it into a real object
94  }
95 
96  if(Value.GetType() == typeof(JArray)) // For when an array appears
97  {
98  var JsonArray = new ScriptArray();
99  foreach(var subElement in prop.Value.ToObject<List<object>>())
100  {
101  if(subElement.GetType() == typeof(JObject)) // JSON within JSON fails to convert to C# object using ToObject<object>, must be done explicitly
102  {
103  JsonArray.Add(JObject.Parse(subElement.ToString()).ToObject<Dictionary<string, object>>()); // Convert JObject in JSON to C# Dictionary
104  }
105  else
106  {
107  JsonArray.Add(subElement);
108  }
109  }
110  obj[prop.Name] = JsonArray;
111  }
112  else
113  {
114  obj[prop.Name] = Value;
115  }
116  }
117  return obj;
118  default:
119  return null;
120  }
121  }

174  public static void AppendEntry(string Collection, JObject Entry)
175  {
176  // Get Collection data, deserialize entries into List of JObjects
177  JObject CollectionObject = JObject.Parse(File.ReadAllText("./_" + Collection + "/collection.json"));
178  List<JObject> Entries = JsonConvert.DeserializeObject<List<JObject>>(CollectionObject["entries"].ToString(Formatting.None));
179  Entries.Add(Entry); // Add current entry to List
180 
181  try
182  {
183  // Sort by date, then title
184  Entries.Sort((y, x) => x["date"].ToString().CompareTo(y["date"].ToString()));
185  Entries.Sort((y, x) => x["title"].ToString().CompareTo(y["title"].ToString()));
186  }
187  catch(NullReferenceException)
188  {
189  Entries.Sort((y, x) => x["title"].ToString().CompareTo(y["title"].ToString()));
190  }
191  catch(InvalidOperationException)
192  {
193  // If page is missing both `date` and `title`
194  }
195 
196  var DeserializedCollection = JsonConvert.DeserializeObject<Dictionary<string, object>>(CollectionObject.ToString(Formatting.None));
197  DeserializedCollection["entries"] = Entries.ToArray(); // Add List into JSON
198 
199  File.WriteAllText("./_" + Collection + "/collection.json", JsonConvert.SerializeObject(DeserializedCollection));
200  }

251  public static string RenderPage(JObject PageObject, string Collection, bool RenderWithLayout, bool isMarkdown, JObject PaginatorObject)
252  {
253  string PageContents = PageObject["content"].ToString();
254  if(isMarkdown)
255  {
256  // Turn Markdown into HTML
257  var builder = new MarkdownPipelineBuilder().UsePipeTables().UseEmphasisExtras().UseAutoLinks().UseTaskLists().UseListExtras().UseMediaLinks().UseMathematics().UseDiagrams();
258  builder.BlockParsers.TryRemove<IndentedCodeBlockParser>();
259  var pipeline = builder.Build();
260  PageContents = Markdown.ToHtml(Markdown.Normalize(PageContents, new NormalizeOptions() { }, pipeline), pipeline);
261  }
262 
263  // Get outside data
264  JObject Global = JObject.Parse(GetGlobal());
265  Global.Merge(GetCollections(), new JsonMergeSettings { MergeArrayHandling = MergeArrayHandling.Union });
266  JObject CollectionObject = new JObject {};
267  if(PageObject.ContainsKey("collection"))
268  {
269  // Can't use `Collection` parameter, as sometimes we don't want the output filed as an entry for said collection
270  CollectionObject = JObject.Parse(File.ReadAllText("./_" + PageObject["collection"].ToString() + "/collection.json"));
271  }
272 
273  // If a page has a layout, use it
274  if(PageObject.ContainsKey("layout") && RenderWithLayout)
275  {
276  PageContents = Layouts[PageObject["layout"].ToString()].Replace("{{ content }}", PageContents);
277  }
278 
279  // Render with Scriban
280  var template = Template.ParseLiquid(PageContents);
281  PageContents = template.Render(new { page = ConvertFromJson(PageObject), collection = ConvertFromJson(CollectionObject), global = ConvertFromJson(Global), pigmeat = ConvertFromJson(GetPigmeat()), paginator = ConvertFromJson(PaginatorObject) });
282  PageContents = Snippet.Parse(PageContents, PageObject); // Parse for snippets
283 
284  if(!string.IsNullOrEmpty(Collection))
285  {
286  PageObject["content"] = RenderPage(PageObject, "", false, isMarkdown, new JObject {}); // TODO: Figure out a way not to have to do this. Inefficient!
287  if(!Serving)
288  {
289  IO.AppendEntry(Collection, PageObject); // When serving we don't want to duplicate entries
290  }
291  }
292  return PageContents;
293  }

304  public static string GetLayoutContents(string LayoutPath, bool Overwrite)
305  {
306  var SplitPage = Page.SplitFrontmatter(File.ReadAllText(LayoutPath));
307  string PageFrontmatter = SplitPage[0];
308  string LayoutContents = SplitPage[1];
309  string Layout = Path.GetFileNameWithoutExtension(LayoutPath);
310  try
311  {
312  string SubLayout = IO.GetYamlObject(PageFrontmatter)["layout"].ToString();
313 
314  if(!Layouts.ContainsKey(SubLayout))
315  {
316  LayoutContents = GetLayoutContents(("./layouts/" + SubLayout + ".html").Replace("{{ content }}", LayoutContents), false);
317  }
318  else
319  {
320  LayoutContents = Layouts[SubLayout].Replace("{{ content }}", LayoutContents);
321  }
322 
323  if(!Layouts.ContainsKey(Layout)) // Some systems may process layout files in an order which will cause an exception
324  {
325  Layouts.Add(Layout, LayoutContents);
326  }
327  else if(Overwrite == true) // For if a layout is changed during 'serve'
328  {
329  Layouts[Layout] = LayoutContents;
330  }
331 
332  return LayoutContents;
333  }
334  catch(Exception)
335  {
336  if(!Layouts.ContainsKey(Layout)) // See above
337  {
338  Layouts.Add(Layout, LayoutContents);
339  }
340  return LayoutContents;
341  }
342  }

349  public static void IncludeDirectory(string SourceDirectory, string DestinationDirectory)
350  {

353  Directory.CreateDirectory(DestinationDirectory);
