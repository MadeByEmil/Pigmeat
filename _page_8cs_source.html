89  public static JObject GetPageObject(string PagePath)
90  {
91  var SplitPage = Page.SplitFrontmatter(File.ReadAllLines(PagePath));
92  string PageFrontmatter = SplitPage[0];
93 
94  JObject FrontmatterObject = IO.GetYamlObject(PageFrontmatter);
95  Page page = JsonConvert.DeserializeObject<Page>(FrontmatterObject.ToString(Formatting.None));
96 
97  page.name = Path.GetFileNameWithoutExtension(PagePath);
98  page.dir = Path.GetDirectoryName(PagePath);
99  page.content = SplitPage[1];
100 
101  JObject GlobalObject = JObject.Parse(IO.GetGlobal());
102  GlobalObject.Merge(JObject.Parse(IO.GetCollections().ToString(Formatting.None)), new JsonMergeSettings { MergeArrayHandling = MergeArrayHandling.Union });
103  CultureInfo Culture = new CultureInfo(GlobalObject["culture"].ToString()); // Get 'culture' from Global
104  page.year = page.date.ToString("yyyy", Culture);
105  page.short_year = page.date.ToString("y", Culture);
106  page.month = page.date.ToString("MM", Culture);
107  page.i_month = page.date.ToString("M", Culture);
108  page.short_month = page.date.ToString("MMM", Culture);
109  page.long_month = page.date.ToString("MMMM", Culture);
110  page.day = page.date.ToString("dd", Culture);
111  page.i_day = page.date.ToString("d", Culture);
112  page.y_day = GetDayOfYear(page.date, Culture);
113  page.week = Culture.Calendar.GetWeekOfYear(page.date, Culture.DateTimeFormat.CalendarWeekRule, Culture.DateTimeFormat.FirstDayOfWeek).ToString(Culture);
114  page.w_day = (int) Culture.Calendar.GetDayOfWeek(page.date);
115  page.w_year = ISOWeek.GetYear(page.date);
116  page.short_day = Culture.DateTimeFormat.GetAbbreviatedDayName(Culture.Calendar.GetDayOfWeek(page.date));
117  page.long_day = Culture.DateTimeFormat.GetDayName(Culture.Calendar.GetDayOfWeek(page.date));
118  page.hour = page.date.ToString("HH", Culture);
119  page.minute = page.date.ToString("mm", Culture);
120  page.second = page.date.ToString("ss", Culture);
121 
122  JObject PageObject = JObject.Parse(JsonConvert.SerializeObject(page, Formatting.None));
123  PageObject.Merge(JObject.Parse(FrontmatterObject.ToString(Formatting.None)), new JsonMergeSettings { MergeArrayHandling = MergeArrayHandling.Union });
124 
125  // Get URL from collected page data, merge back into JObject
126  page.url = GetPermalink(PageObject, new JObject {}); // Generate output path based on other variables
127  PageObject.Merge(JObject.Parse(JsonConvert.SerializeObject(page, Formatting.None)), new JsonMergeSettings { MergeArrayHandling = MergeArrayHandling.Union });
128 
129  try
130  {
131  string DirectoryName = Path.GetDirectoryName(PagePath);
132  bool isMarkdown = Path.GetExtension(PagePath).Equals(".md");
133  if(!DirectoryName.Equals(".") && DirectoryName.Substring(0, 3).Equals("./_"))
134  {
135  page.content = IO.RenderPage(PageObject, DirectoryName.Substring(3), true, isMarkdown, new JObject {});
136  }
137  else
138  {
139  page.content = IO.RenderPage(PageObject, "", true, isMarkdown, new JObject {});
140  }
141  }
142  catch(ArgumentOutOfRangeException) // If not in root, but not in a collection
143  {
144  bool isMarkdown = Path.GetExtension(PagePath).Equals(".md");
145  page.content = IO.RenderPage(PageObject, "", true, isMarkdown, new JObject {});
146  }
147 
148  // Merge content back into JObject
149  PageObject.Merge(JObject.Parse(JsonConvert.SerializeObject(page, Formatting.None)), new JsonMergeSettings { MergeArrayHandling = MergeArrayHandling.Union });
150  return PageObject; // Return JObject of page
151  }

162  public static string[] SplitFrontmatter(string[] PageLines)
163  {
164  bool BeginFrontmatter = false;
165  bool EndFrontmatter = false;
166  StringBuilder Frontmatter = new StringBuilder();
167  StringBuilder Contents = new StringBuilder();
168 
169  foreach(var line in PageLines)
170  {
171  var CurrentLine = line;
172  if(!BeginFrontmatter && CurrentLine.Equals("---"))
173  {
174  BeginFrontmatter = true;
175  }
176  else if(BeginFrontmatter && CurrentLine.Equals("---") && !EndFrontmatter)
177  {
178  EndFrontmatter = true;
179  }
180  else if(BeginFrontmatter && !EndFrontmatter)
181  {
182  Frontmatter.AppendLine(CurrentLine);
183  }
184  else
185  {
186  Contents.AppendLine(CurrentLine);
187  }
188  }
189 
190  return new string[] { Frontmatter.ToString(), Contents.ToString() };
191  }

202  public static string GetPermalink(JObject PageObject, JObject PaginatorObject)
203  {
204  string Permalink = PageObject["permalink"].ToString();
205  switch(Permalink)
206  {
207  case "date":
208  Permalink = "/{{ page.collection }}/{{ page.year }}/{{ page.month }}/{{ page.day }}/{{ page.title }}.html";
209  break;
210  case "pretty":
211  Permalink = "/{{ page.collection }}/{{ page.year }}/{{ page.month }}/{{ page.day }}/{{ page.title }}.html";
212  break;
213  case "ordinal":
214  Permalink = "/{{ page.collection }}/{{ page.year }}/{{ page.y_day }}/{{ page.title }}.html";
215  break;
216  case "weekdate":
217  Permalink = "/{{ page.collection }}/{{ page.year }}/W{{ page.week }}/{{ page.short_day }}/{{ page.title }}.html";
218  break;
219  case "none":
220  Permalink = "/{{ page.collection }}/{{ page.title }}.html";
221  break;
222  default:
223  break;
224  }
225  var template = Template.ParseLiquid(Permalink);
226  return template.Render(new { page = PageObject, global = IO.GetGlobal(), pigmeat = IO.GetPigmeat(), paginator = PaginatorObject });
227  }
228  static string GetDayOfYear(DateTime Date, CultureInfo Culture)
229  {
230  if(Date.DayOfYear.ToString().Length == 1)

233  }
234  else if(Date.DayOfYear.ToString().Length == 2)
235  {
236  return "0" + Date.DayOfYear.ToString(Culture);
237  }
238  else
239  {
240  return Date.DayOfYear.ToString(Culture);
241  }
242  }
