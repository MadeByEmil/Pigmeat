48  public static string Parse(string Contents, JObject PageObject)
49  {
50  if(Contents.Contains("{! snippet "))
51  {
52  List<string> SnippetCalls = new List<string>();
53  StringBuilder Reader = new StringBuilder();
54  int BraceCount = 0;
55  foreach(var character in Contents)
56  {
57  if(character.Equals('{') && BraceCount == 0)
58  {
59  BraceCount++;
60  Reader.Append(character);
61  continue;
62  }
63  if(character.Equals('}') && BraceCount == 1)
64  {
65  BraceCount = 0;
66  Reader.Append(character);
67  var ReaderString = Reader.ToString();
68  if(ReaderString.Contains("{! snippet ") || ReaderString.Contains("{!snippet "))
69  {
70  SnippetCalls.Add(WebUtility.HtmlDecode(ReaderString));
71  }
72  Reader = new StringBuilder();
73  continue;
74  }
75  if(BraceCount == 1)
76  {
77  Reader.Append(character);
78  continue;
79  }
80  }
81  foreach(var snippetCall in SnippetCalls)
82  {
83  Variables = new Dictionary<string, string>(); // Reset variable data for each new call
84  Regex CallSplit = new Regex("=| (?=(\\w*)+=)"); // Regex to split based on space (not in quotes) and equals sign
85  // Do the split, remove bits of call syntax, trim, remove empty values created by the split
86  string[] CallArguments = CallSplit.Split(snippetCall.Replace("snippet ", "").Replace("{!", "").Replace("!}", "").Trim()).Where(x => !string.IsNullOrEmpty(x)).ToArray();
87  List<string> Keys = new List<string>();
88  List<string> Values = new List<string>();
89  for(int i = 0; i < CallArguments.Length - 1; i++) // Offset of one to ignore snippet filename
90  {
91  switch(i % 2)
92  {
93  case 0:
94  Keys.Add(CallArguments[i + 1]);
95  break;
96  case 1:
97  Values.Add(CallArguments[i + 1]);
98  break;
99  }
100  }
101  for(int i = 0; i < Keys.ToArray().Length; i++)
102  {
103  Variables.Add(Keys[i], Values[i]); // Merge `List`s into `Dictionary`
104  }
105  string SnippetPath = "./snippets/" + CallArguments[0]; // Trimming required for this to work
106  Contents = Contents.Replace(WebUtility.HtmlEncode(snippetCall), Snippet.Render(SnippetPath, PageObject));
107  }
108  }
109  return Contents;
110  }
