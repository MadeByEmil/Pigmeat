45  public static string Parse(string Contents, JObject PageObject)
46  {
47  if(Contents.Contains("{! snippet "))
48  {
49  List<string> SnippetCalls = new List<string>();
50  string ReaderString = "";
51  bool HitFirstBrace = false;
52  foreach(var character in Contents)
53  {
54  if(character.Equals('{') && !HitFirstBrace)
55  {
56  HitFirstBrace = true;
57  ReaderString += character;
58  continue;
59  }
60  if(character.Equals('}') && HitFirstBrace)
61  {
62  HitFirstBrace = false;
63  ReaderString += character;
64  if(ReaderString.Contains("{! snippet "))
65  {
66  SnippetCalls.Add(ReaderString);
67  }
68  ReaderString = "";
69  continue;
70  }
71  if(HitFirstBrace)
72  {
73  ReaderString += character;
74  continue;
75  }
76  }
77  foreach(var snippetCall in SnippetCalls)
78  {
79  Snippet CurrentSnippet = new Snippet { Input = snippetCall };
80  string SnippetPath = "./snippets/" + CurrentSnippet.GetArguments()[2];
81  Contents = Contents.Replace(snippetCall, CurrentSnippet.Render(SnippetPath, PageObject));
82  }
83  }
84  return Contents;
85  }

155  string[] GetValues()
156  {
157  List<string> Values = new List<string>();
158  string CurrentValue = "";
159  for(int i = 3; i < GetArguments().Length; i++)
160  {
161  bool HitEquals = false; // Whether or not we've reached the equals sign yet
162  foreach(var character in GetArguments()[i])
163  {
164  if(character.Equals('='))
165  {
166  HitEquals = true;
167  continue;
168  }
169  if(HitEquals)
170  {
171  CurrentValue += character;
172  continue;
173  }
174  }
175  try
176  {
177  if(CurrentValue.ToCharArray()[0].Equals('"') && CurrentValue.ToCharArray()[CurrentValue.Length - 1].Equals('"'))
178  {
179  Values.Add(CurrentValue.Substring(1, CurrentValue.Length - 2));
180  }
181  else
182  {
183  Values.Add(CurrentValue);
184  }
185  }
186  catch(IndexOutOfRangeException)

191  }
