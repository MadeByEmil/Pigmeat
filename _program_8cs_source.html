102  case "serve":

336  static void ServeBuild(object source, FileSystemEventArgs e)
337  {
338  var RelativePath = Path.GetRelativePath("./", e.FullPath); // Relative paths are nicer to work with
339  /*
340  Filter out the following situations:
341  1. Changes by Pigmeat iself
342  2. Changes recorded by git
343  3. Changes to directories (no purpose in doing anything when this occurs)
344  4. Changes made mere milliseconds apart
345  */
346  if(File.GetAttributes(e.FullPath).HasFlag(FileAttributes.Directory) || Path.GetFileName(RelativePath).Equals("_global.yml") || RelativePath.Contains("output/") || RelativePath.Contains(".git/") || DateTime.Now.Subtract(LastFileWatcherEvent).TotalMilliseconds < 500)
347  {
348  return;
349  }
350  LastFileWatcherEvent = DateTime.Now; // Keeps track of when the last proper event occured
351 
352  Console.WriteLine("\n" + RelativePath + " has been modified. Rebuilding … ");
353 
354  JObject Global = JObject.Parse(IO.GetGlobal());
355  string[] IncludedFiles = JsonConvert.DeserializeObject<string[]>(Global["included-files"].ToString());
356  string[] IncludedPages = JsonConvert.DeserializeObject<string[]>(Global["included-pages"].ToString());
357 
358  // If a layout is changed, overwrite it in the cache and rebuild its dependencies
359  if(Path.GetDirectoryName(RelativePath).Equals("layout"))
360  {
361  IO.GetLayoutContents(RelativePath, true);
362  foreach(var file in Directory.GetFiles("./", "*", SearchOption.TopDirectoryOnly))
363  {
364  if(Path.GetExtension(RelativePath).Equals(".md") || Path.GetExtension(RelativePath).Equals(".html"))
365  {
366  JObject PageObject = Page.GetPageObject(RelativePath);
367  if(PageObject["layout"].ToString().Equals(Path.GetFileNameWithoutExtension(RelativePath)))
368  {
369  Directory.CreateDirectory(Path.GetDirectoryName("./output/" + PageObject["url"].ToString()));
370  File.WriteAllText("./output/" + PageObject["url"].ToString(), PageObject["content"].ToString());
371  }
372  }
373  }
374  }
375  // If a snippet is changed, rebuild its dependencies
376  else if(Path.GetDirectoryName(RelativePath).Equals("snippets"))
377  {
378  foreach(var file in Directory.GetFiles("./", "*", SearchOption.TopDirectoryOnly))
379  {
380  if(Path.GetExtension(RelativePath).Equals(".md") || Path.GetExtension(RelativePath).Equals(".html"))
381  {
382  if(File.ReadAllText(file).Contains("{! snippet " + Path.GetFileName(RelativePath)))
383  {
384  JObject PageObject = Page.GetPageObject(RelativePath);
385  Directory.CreateDirectory(Path.GetDirectoryName("./output/" + PageObject["url"].ToString()));
386  File.WriteAllText("./output/" + PageObject["url"].ToString(), PageObject["content"].ToString());
387  }
388  }
389  }
390  }
391  // If a document is changed, rebuild it
392  else if(Path.GetExtension(RelativePath).Equals(".md") || Path.GetExtension(RelativePath).Equals(".html"))
393  {
394  JObject PageObject = Page.GetPageObject(RelativePath);
395  Directory.CreateDirectory(Path.GetDirectoryName("./output/" + PageObject["url"].ToString()));
396  File.WriteAllText("./output/" + PageObject["url"].ToString(), PageObject["content"].ToString());
397  }
398  // If the stylesheet(s) are changed, rebuild them
399  else if(Path.GetExtension(RelativePath).Equals(".scss") || Path.GetExtension(RelativePath).Equals(".sass"))
400  {
401  Directory.CreateDirectory("./output/" + Path.GetDirectoryName(RelativePath));
402  File.WriteAllText("./output/" + Path.GetDirectoryName(RelativePath) + "/" + Path.GetFileNameWithoutExtension(RelativePath) + ".css", Scss.ConvertToCss(File.ReadAllText(RelativePath)).Css);
403  }
404  // If a file has changed, copy it again
405  else if(!Path.GetExtension(RelativePath).Equals(".json") && !Path.GetExtension(RelativePath).Equals(".yml"))
406  {
407  Directory.CreateDirectory("./output/" + Path.GetDirectoryName(RelativePath));
408  File.Copy(RelativePath, "./output/" + RelativePath, true);
409  }
410  // If 'included' and changed, then re-output
411  try
412  {
413  foreach(var file in IncludedFiles)
414  {
415  if(file.Equals(RelativePath))
416  {
417  Directory.CreateDirectory("./output/" + Path.GetDirectoryName(RelativePath));
418  try
419  {
420  File.Copy(RelativePath, "./output/" + RelativePath, true);
421  }
422  catch(IOException)
423  {
424  IO.IncludeDirectory(RelativePath, "./output/" + RelativePath);
425  }
426  }
427  }
428  }
429  catch(Exception ex)
430  {
431  Console.WriteLine(ex);
432  }
433  try
434  {
435  foreach(var file in IncludedPages)
436  {
437  if(file.Equals(RelativePath))
438  {
439  JObject PageObject = Page.GetPageObject(RelativePath);
440  Directory.CreateDirectory(Path.GetDirectoryName("./output/" + PageObject["url"].ToString()));
441  File.WriteAllText("./output/" + PageObject["url"].ToString(), PageObject["content"].ToString());
442  }
443  }
444  }
445  catch(Exception ex)
446  {
447  Console.WriteLine(ex);
448  }
449  return;
450  }
